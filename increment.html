async function fetchInitialValue() { try { const response = await fetch('/countervalue'); if (response.ok) { const data = await response; counterValue = data; document.getElementById("counter").innerText = counterValue; } else { console.error('Failed to fetch initial value'); } } catch (error) { console.error('Error:', error); } } // Fetch and set the initial value when the page loads window.onload = function() { fetchInitialValue(); };
