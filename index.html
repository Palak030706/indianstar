import os # List of file paths to include in the HTML file_paths = [ 'abcd.html', 'palak.html', 'README.md' ] # Function to read file contents def read_file(file_path): with open(file_path, 'r') as file: return file.read() # Function to generate HTML content def generate_html(file_paths): html_content = "\n\nFiles Content\n\n\n" for file_path in file_paths: if os.path.exists(file_path): file_content = read_file(file_path) html_content += f"

Contents of {file_path}

\n
{file_content}
\n" else: html_content += f"

Contents of {file_path}

\n

File not found.

\n" html_content += "\n" return html_content # Generate HTML content html_output = generate_html(file_paths) # Write HTML content to a file output_file_path = 'output.html' with open(output_file_path, 'w') as output_file: output_file.write(html_output) print(f"HTML file generated: {output_file_path}")
